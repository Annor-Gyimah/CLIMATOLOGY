import xarray as xr
import matplotlib.pyplot as plt

# define file path
file_path = 'daily_rainfall.nc' # replace with your own file path

# read in data as an xarray Dataset
ds = xr.open_dataset(file_path)

# select region of interest
lon_min, lon_max = -10, 10
lat_min, lat_max = 0, 20
ds_roi = ds.sel(lon=slice(lon_min, lon_max), lat=slice(lat_min, lat_max))

# calculate annual totals and average
annual_totals = ds_roi['rainfall'].resample(time='1AS').sum(dim='time')
annual_average = ds_roi['rainfall'].resample(time='1AS').mean(dim='time')

# create time-series plot of annual totals and average
fig, ax = plt.subplots()
annual_totals.plot(ax=ax, label='Annual Totals')
annual_average.plot(ax=ax, label='Annual Average')
ax.set(xlabel='Year', ylabel='Rainfall (mm)', title='Rainfall Totals and Average per Year')
ax.legend()

plt.show()
